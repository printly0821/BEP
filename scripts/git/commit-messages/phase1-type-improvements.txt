### 타입 정의
```typescript
export type VariableCostDetail = {
  materials?: number;      // 원재료비
  packaging?: number;      // 패키지
  shippingBox?: number;    // 택배박스
  marketFee?: number;      // 마켓수수료
  shippingCost?: number;   // 배송비
  other?: number;          // 기타
};

export type FixedCostDetail = {
  labor?: number;          // 인건비
  meals?: number;          // 식비
  rent?: number;           // 임대료
  utilities?: number;      // 공과금
  office?: number;         // 사무실운영비
  marketing?: number;      // 마케팅비
  other?: number;          // 기타
};

export type CalculationInputs = {
  price: number;
  unitCost: number;                           // 변동비 합계 (필수)
  fixedCost: number;                          // 고정비 합계 (필수)
  targetProfit?: number;
  variableCostDetail?: VariableCostDetail;    // 세부 항목 (선택)
  fixedCostDetail?: FixedCostDetail;          // 세부 항목 (선택)
};
```
